def insert_fin(val)
        aux = Nodo.new(nil,val,nil)
        if empty()
                @inicio = aux
        else
                @final.sig = aux
                aux.prev = @final
        end
        @final = aux
end

def to_s()
        s = ""
        aux = @inicio
        while !aux.nil?
                s << aux.valor.to_s
                s << "\n"
                aux = aux.sig
        end
        
        s
end
