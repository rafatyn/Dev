def to_s
        s = "#{@tit} (#{@ing[0]}-#{@ing[1]})\n"
        i=0
        while i<@platos.length
                s << "- #{@platos[i][0]}, #{@platos[i][1]}, #{@platos[i][2]}\n"
                i+=1
        end
        s << "V.C.T | %\t#{@porc[0]} kcal | #{@porc[1]}% - #{@porc[2]}% - #{@porc[3]}%\n" 
end
